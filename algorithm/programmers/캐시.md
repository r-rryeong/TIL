def solution(cacheSize, cities):
    answer = 0
    q = []
    
    if cacheSize==0:
        answer = len(cities) * 5
    else:
        for c in cities:
            c = c.lower()
            if c in q:
                q.remove(c)
                q.append(c)
                answer += 1
            else:
                if len(q) >= cacheSize:
                    q.pop(0)
                q.append(c)
                answer += 5
        
    return answer
