```python
def solution(n):
    answer = ''
    array = ['1', '2', '4']

    # 규칙!
    # n//3의 124나라 숫자
    #     +
    # (n-1)%3: 일의 자리 숫자의 array 인덱스

    if n <= 3:   # 3이하의 숫자는 array 차례대로 출력
        answer = array[(n - 1) % 3]
    else:
        answer = array[(n - 1) % 3]
        n //= 3
        answer = array[(n - 1) % 3] + answer

    return answer
```
