def solution(m, n, board):
    for b in range(m):
        board[b] = list(board[b])
        
    answer = 0
    visited = set()
    for r in range(m-1):
        for c in range(n-1):
            if board[r][c]==board[r][c+1]==board[r+1][c]==board[r+1][c+1]:
                visited.add((r,c))
                visited.add((r,c+1))
                visited.add((r+1,c))
                visited.add((r+1,c+1))
    
    return answer
