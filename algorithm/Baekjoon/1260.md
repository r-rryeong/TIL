### DFS와 BFS

```python
N, M, V = map(int, input().split())
# N 정점의 개수, M 간선의 개수, V 탐색 시작 정점 번호
edge = [list(map(int, input().split())) for _ in range(M)]
# 간선이 연결하는 두 정점의 번호
graph = [[0]*(N+1) for _ in range(N+1)]

# 정점과 간선을 나타내는 그래프 작성
for x, y in edge:
    graph[x][y] = 1
    graph[y][x] = 1

# DFS
visited = [0] * (N+1)
visited[V] = 1
stack = []
path = [V]   # 방문한 정점
i = V
while 1:
    for j in range(1, N+1):
        if graph[i][j] and not visited[j]:
            stack.append(i)
            path.append(j)
            visited[j] = 1
            i = j   # 방문할 정점 갱신
            break
    else:
        if stack:
            i = stack.pop()
        else:
            break
print(*path)
# stack = []
# path = [1, 2, 4, 3]

# BFS
visited = [0] * (N+1)
visited[V] = 1
queue = [V]
path = [V]
while queue:
    i = queue.pop(0)
    for j in range(1, N+1):
        if graph[i][j] and not visited[j]:
            queue.append(j)
            path.append(j)
            visited[j] = 1
print(*path)
```
