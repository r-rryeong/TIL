import sys
from collections import deque
input = sys.stdin.readline

def bfs(sr, sc, visited):
    global c_o, c_v
    q = deque()
    q.append((sr, sc))
    if arr[sr][sc]=='o':
        c_o += 1
    else:
        c_v += 1
    visited[sr][sc] = 1

    while q:
        cr, cc = q.popleft()
        for dr, dc in (-1, 0), (0, 1), (1, 0), (0, -1):
            nr = cr + dr
            nc = cc + dc
            if 0<=nr<R and 0<=nc<C and not visited[nr][nc] and arr[nr][nc]!='#':
                visited[nr][nc]=1
                q.append((nr, nc))
                if arr[nr][nc]=='o':
                    c_o += 1
                elif arr[nr][nc]=='v':
                    c_v += 1


R, C = map(int, input().split())
arr = []
for _ in range(R):
    a = list(input())
    arr.append(a)

res1 = res2 = 0
visited = [[0]*C for _ in range(R)]
for r in range(R):
    for c in range(C):
        if (arr[r][c]=='o' or arr[r][c]=='v') and visited[r][c]==0:
            c_o = c_v = 0
            bfs(r, c, visited)
            if c_o > c_v:
                c_v = 0
            else:
                c_o = 0
            res1 += c_o
            res2 += c_v

print(res1, res2)
