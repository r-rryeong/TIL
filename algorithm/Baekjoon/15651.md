### N과 M(3)

```python
# 1.
import itertools

N, M = map(int, input().split())
num_list = [i for i in range(1, N+1)]

c_num = itertools.product(num_list, repeat=M)
for i in c_num:
    for j in i:
        print(j, end = ' ')
    print()
```

```python
# 2.
import sys
sys.setrecursionlimit(10**8)    # 최대 재귀 깊이 늘려주기

N, M = map(int, input().split())
numbers = [i for i in range(1, N+1)]
stack = []

def dfs():
    if len(stack) == M:
        print(' '.join(map(str, stack)))
        return

    for i in numbers:
        stack.append(i)
        dfs()
        stack.pop()

dfs()
```
