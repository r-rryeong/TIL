### 폴짝폴짝

```python
import sys
from collections import deque
input = sys.stdin.readline

def bfs(s, e, N):
    q = deque([s-1])
    check = [-1]*1000
    check[s-1] = 0

    while q:
        v = q.popleft()
        for n in range(v, N, bridge[v]):
            if check[n] == -1:
                q.append(n)
                check[n] = check[v]+1
                if n==e-1:
                    return check[n]

        for n in range(v, -1, -bridge[v]):
            if check[n]==-1:
                q.append(n)
                check[n] = check[v]+1
                if n==e-1:
                    return check[n]

    return -1


N = int(input())   # 징검다리의 개수
bridge = [0]+list(map(int, input().split()))
a, b = map(int, input().split())   # a번~b번 징검다리

res = bfs(a, b, N)

print(res)
```
