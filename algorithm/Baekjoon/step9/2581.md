### 소수
```python
m = int(input())
n = int(input())

number = []
for num in range(m, n+1):
    number.append(num)

s_lst = []
for i in number:
    not_s = 0
    if i > 1:
        for j in range(2, i):
            if i % j == 0:
                not_s += 1
                break
        if not_s == 0:
            s_lst.append(i)

if s_lst == []:
    print(-1)
else:
    total = sum(s_lst)
    s_min = min(s_lst)
    print(total)
    print(s_min)
```
