### 수 묶기

```python
# 미완성
N = int(input())
arr = []
for _ in range(N):
    arr.append(int(input()))

lst = []
plus = []
minus = []
for n in arr:
    if n > 0:
        if n==1:
            lst.append(n)
        else:
            plus.append(n)
    else:
        minus.append(n)

if plus:
    plus.sort(reverse=True)
    # print(plus)
    if len(plus) % 2:
        for p in range(0, len(plus)-1, 2):
            lst.append(plus[p]*plus[p+1])
            plus.pop(p)
            plus.pop(p+1)
        while plus:
            lst.append(plus.pop())
    else:
        for p in range(0, len(plus), 2):
            lst.append(plus[p]*plus[p+1])
            if p == len(plus)//2:
                break

if len(minus) >= 2:
    minus.sort()
    if len(minus) % 2:
        for m in range(0, len(minus)-1, 2):
            lst.append(minus[m]*minus[m+1])
            minus.pop(m)
            minus.pop(m+1)
        while minus:
            lst.append(minus.pop())
    else:
        for m in range(0, len(minus), 2):
            lst.append(minus[m]*minus[m+1])
            if p == len(minus)//2:
                break
else:
    lst.append(minus[0])
# print(lst)
print(sum(lst))
```
