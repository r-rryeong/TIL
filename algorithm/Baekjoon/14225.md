### 부분수열의 합

```python
import sys

input = sys.stdin.readline

def powerset(n, k, ssum):
    global sum_lst
    if k==n:
        sum_lst[ssum] = 1
        return
    else:
        powerset(n, k+1, ssum + S[k])
        powerset(n, k+1, ssum)

N = int(input())
S = list(map(int, input().split()))
end_idx = sum(S)+1
sum_lst = [0] * (end_idx+1)
powerset(N, 0, 0)

for x in range(1, end_idx+1):
    if not sum_lst[x]:
        print(x)
        break
```
