```python
# 1. itertools

from itertools import combinations

while True:
    lst = list(map(int, input().split()))
    k = lst.pop(0)
    if k==0:
        break
    for i in combinations(lst, k):
        A = list(i)
        for j in combinations(A, 6):
            print(*j)
    print(end='\n')
```
