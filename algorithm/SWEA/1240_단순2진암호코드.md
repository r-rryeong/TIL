```python
import sys

sys.stdin = open('input.txt', 'r')

T = int(input())
for tc in range(1, T+1):
    N, M = map(int, input().split())   # 배열의 세로 크기, 가로 크기
    arr = [input() for _ in range(N)]
    # 암호코드
    pattern = {
        '0001101': 0,
        '0011001': 1,
        '0010011': 2,
        '0111101': 3,
        '0100011': 4,
        '0110001': 5,
        '0101111': 6,
        '0111011': 7,
        '0110111': 8,
        '0001011': 9,
    }

    for r in range(N):
        i = M
        code = []
        while i > 0:
            b = arr[r][i-7: i]
            if b in pattern:
                code.insert(0, pattern[b])
                i -= 7
            else:
                i -= 1
        if code:
            break

    calc = 0
    for o in range(8):
        if (o+1) % 2:
            calc += code[o]
    calc *= 3
    for e in range(8):
        if (e+1) % 2 == 0:
            calc += code[e]

    res = 0
    if calc % 10 == 0:
        res = sum(code)
    else:
        res = 0
    print(f'#{tc}', res)
```
