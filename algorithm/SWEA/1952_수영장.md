def DFS(n, ssum):
    global ans
    if n>12:
        if ans>ssum:
            ans = ssum
        return

    DFS(n+1, ssum+lst[n]*day)
    DFS(n+1, ssum+mon)
    DFS(n+3, ssum+mon3)
    DFS(n+12, ssum+year)

T = int(input())
for tc in range(1, T+1):
    day, mon, mon3, year = map(int, input().split())
    lst = [0] + list(map(int, input().split()))
    ans = 123454345
    DFS(1, 0)
    print(f'#{tc} {ans}')
